def self.create_delegated_method(klass, name)
  method = klass.instance_method(name)

  redefine_method(klass, name) do |*args, &block|
    PolyDelegate.delegate_call(self, method, *args, &block)
  end
end

def self.delegate_call(obj, method, *args, &block)
  delegator, args = distinguish_delegator(obj, *args)
  bound = method.force_bind(delegator)
  bound.call(*args, &block)
end

def self.distinguish_delegator(obj, *args)
  delegator =
    if args.empty? || !args.first.is_a?(Delegator)
      obj
    else
      args.shift
    end

  [delegator, args]
end

def self.method_visibility(mod, method)
  if mod.public_method_defined? method
    :public
  elsif mod.private_method_defined? method
    :private
  elsif mod.protected_method_defined? method
    :protected
  end
end

def self.redefine_method(mod, name, &block)
  visibility = method_visibility(mod, name)

  # Use remove_method instead of undef_method. The difference is minor,
  # but remove_method doesn't replace the method with a fake method.
  mod.__send__(:remove_method, name)
  mod.__send__(:define_method, name, &block)

  set_method_visibility(mod, name, visibility)
end
